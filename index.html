function cancelAdmission() { if(!confirm("શું તમે આ પ્રવેશ રદ કરવા માંગો છો?")) return; const cancelBtn = document.getElementById('cancelBtn'); const redirectMsg = document.getElementById('redirectMsg'); // બટન ફ્રીઝ કરો અને મેસેજ બતાવો cancelBtn.disabled = true; cancelBtn.style.opacity = "0.5"; cancelBtn.innerHTML = "રદ થઈ રહ્યું છે... ⏳"; redirectMsg.style.display = "block"; redirectMsg.innerHTML = "સર્વર સાથે સંપર્ક થઈ રહ્યો છે, કૃપા કરીને રાહ જુઓ..."; const data = JSON.parse(localStorage.getItem("tempData")); var xhr = new XMLHttpRequest(); xhr.open("POST", scriptURL, true); xhr.setRequestHeader("Content-Type", "text/plain;charset=utf-8"); // જો ૧૫ સેકન્ડ સુધી જવાબ ન મળે તો પણ પ્રોસેસ આગળ વધારો xhr.timeout = 15000; xhr.onload = function() { // સફળતાપૂર્વક રદ થયા પછીની પ્રોસેસ finalizeCancellation(); }; xhr.ontimeout = function() { // જો સર્વર મોડું કરે તો પણ લોકલ સ્ટોરેજ સાફ કરી નાખો finalizeCancellation(); }; xhr.onerror = function() { // નેટવર્ક એરર આવે તો પણ યુઝરને અટકાવવાને બદલે ક્લિયર કરો finalizeCancellation(); }; // સર્વરને ડેટા મોકલો xhr.send(JSON.stringify({ studentName: data.studentName, mobile: data.mobile, action: "cancel" })); } // લોકલ સ્ટોરેજ ક્લિયર કરવા અને પેજ રીલોડ કરવાનું કોમન ફંક્શન function finalizeCancellation() { localStorage.removeItem("formSubmitted"); localStorage.removeItem("tempData"); const redirectMsg = document.getElementById('redirectMsg'); document.getElementById('receiptActions').style.display = "none"; redirectMsg.innerHTML = "✅ પ્રવેશ સફળતાપૂર્વક રદ થયો છે. પેજ રીલોડ થઈ રહ્યું છે..."; setTimeout(() => { location.reload(); }, 2000); }
